package com.example.ingredient.parser.dom

import org.junit.jupiter.api.Test
import org.junit.jupiter.api.Assertions.*

class DomBlockScorerTest {

    @Test
    fun `high quality menu block scores above 0_60`() {
        // Per D-03: semanticRole="menu" → semanticRoleScore=1.0
        // Per D-03: rect.y=300 > 200 AND depth=4 in 3..6 → positionScore=1.0
        val block = createBlock(
            priceHits = 5,
            foodHits = 8,
            textLength = 200,
            semanticRole = "menu",
            classes = listOf("menu-items", "food-section"),
            depth = 4,
            rectY = 300f
        )
        
        val result = DomBlockScorer.score(block)
        
        assertTrue(result >= 0.60f, "Menu block should score >= 0.60, got $result")
    }

    @Test
    fun `navbar block scores below 0_20`() {
        val block = createBlock(
            priceHits = 0,
            foodHits = 0,
            textLength = 50,
            semanticRole = "navigation",
            classes = listOf("nav", "main-navigation"),
            depth = 2,
            rectY = 50f
        )
        
        val result = DomBlockScorer.score(block)
        
        assertTrue(result <= 0.20f, "Navbar should score <= 0.20, got $result")
    }

    @Test
    fun `footer block scores below 0_15`() {
        val block = createBlock(
            priceHits = 0,
            foodHits = 0,
            textLength = 100,
            semanticRole = "contentinfo",
            classes = listOf("footer", "site-footer"),
            depth = 2,
            rectY = 800f
        )
        
        val result = DomBlockScorer.score(block)
        
        assertTrue(result <= 0.15f, "Footer should score <= 0.15, got $result")
    }

    @Test
    fun `zero text length does not cause division by zero`() {
        val block = createBlock(
            priceHits = 0,
            foodHits = 0,
            textLength = 0,
            semanticRole = null,
            classes = emptyList(),
            depth = 5,
            rectY = 300f
        )
        
        val result = DomBlockScorer.score(block)
        
        assertTrue(result.isFinite(), "Score should be finite, got $result")
        assertTrue(result >= 0.0f, "Score should be non-negative")
    }

    @Test
    fun `score breakdown weights sum correctly`() {
        // Per D-03: semanticRole="menu" → semanticRoleScore=1.0
        val block = createBlock(
            priceHits = 3,
            foodHits = 5,
            textLength = 100,
            semanticRole = "menu",
            classes = listOf("menu"),
            depth = 4,
            rectY = 300f
        )
        
        val scoredBlock = DomBlockScorer.scoreWithBreakdown(block)
        val breakdown = scoredBlock.breakdown
        
        // Verify weights: 0.30 + 0.25 + 0.20 + 0.15 + 0.10 = 1.0
        val reconstructedScore = 
            0.30f * breakdown.priceDensity +
            0.25f * breakdown.foodDensity +
            0.20f * breakdown.semanticRole +
            0.15f * breakdown.className +
            0.10f * breakdown.position
        
        assertEquals(scoredBlock.score, reconstructedScore, 0.001f,
            "Reconstructed score should match reported score")
    }

    @Test
    fun `block outside good position range has positionScore 0_5`() {
        // Per D-03: positionScore = 1.0 if rect.y > 200 AND depth in 3..6
        //           positionScore = 0.5 otherwise
        // depth=18 is outside 3..6 range → positionScore = 0.5
        val block = createBlock(
            priceHits = 5,
            foodHits = 8,
            textLength = 200,
            semanticRole = "menu",
            classes = listOf("menu"),
            depth = 18,
            rectY = 100f  // Also y <= 200
        )
        
        val scoredBlock = DomBlockScorer.scoreWithBreakdown(block)
        
        assertEquals(0.5f, scoredBlock.breakdown.position, 0.01f,
            "Position score outside good range should be 0.5 per D-03")
    }

    @Test
    fun `block in good position range has positionScore 1_0`() {
        // Per D-03: positionScore = 1.0 if rect.y > 200 AND depth in 3..6
        val block = createBlock(
            priceHits = 0,
            foodHits = 0,
            textLength = 100,
            semanticRole = null,
            classes = emptyList(),
            depth = 4,
            rectY = 300f
        )
        
        val scoredBlock = DomBlockScorer.scoreWithBreakdown(block)
        
        assertEquals(1.0f, scoredBlock.breakdown.position, 0.01f,
            "Position score for rect.y=300, depth=4 should be 1.0 per D-03")
    }

    @Test
    fun `class name matching is case insensitive for menu keywords`() {
        val block = createBlock(
            priceHits = 0,
            foodHits = 0,
            textLength = 100,
            semanticRole = null,
            classes = listOf("MenuItems", "FOOD-list"),
            depth = 5,
            rectY = 300f
        )
        
        val scoredBlock = DomBlockScorer.scoreWithBreakdown(block)
        
        assertEquals(1.0f, scoredBlock.breakdown.className, 0.01f,
            "Class name score should be 1.0 for 'MenuItems' or 'FOOD-list'")
    }

    private fun createBlock(
        priceHits: Int,
        foodHits: Int,
        textLength: Int,
        semanticRole: String?,
        classes: List<String>,
        depth: Int,
        rectY: Float = 0f
    ): DomBlock = DomBlock(
        index = 0,
        tag = "div",
        id = null,
        classes = classes,
        depth = depth,
        childCount = 3,
        textLength = textLength,
        directText = "",
        subtreeText = "Sample text",
        priceHits = priceHits,
        foodHits = foodHits,
        visible = true,
        rect = BlockRect(0f, rectY, 100f, 100f),
        semanticRole = semanticRole,
        schemaType = null,
        xpath = "/html/body/div"
    )
}

package com.example.ingredient.parser.dom

import com.example.ingredient.parser.ParsingMode
import org.junit.jupiter.api.Test
import org.junit.jupiter.api.Assertions.*
import java.io.InputStreamReader

class DomPipelineIntegrationTest {

    @Test
    fun `clean menu fixture selects menu blocks above threshold`() {
        val snapshot = loadFixture("dom_snapshot_clean_menu.json")
        
        val result = MenuBlockSelector.select(snapshot.blocks, ParsingMode.BALANCED)
        
        assertTrue(result.selectedBlocks.isNotEmpty(), "Should select at least one block")
        assertTrue(result.selectedBlocks.size >= 3, "Should select 3+ menu blocks from clean fixture")
        assertTrue(result.selectedBlocks.all { it.score >= 0.25f },
            "All selected blocks should score >= 0.25 (BALANCED threshold)")
    }

    @Test
    fun `noisy page fixture excludes navbar and footer`() {
        val snapshot = loadFixture("dom_snapshot_noisy_page.json")
        
        val result = MenuBlockSelector.select(snapshot.blocks, ParsingMode.BALANCED)
        
        // Verify no navigation or footer blocks selected
        val selectedRoles = result.selectedBlocks.map { it.block.semanticRole }
        assertFalse(selectedRoles.contains("navigation"),
            "Navigation blocks should not be selected")
        assertFalse(selectedRoles.contains("contentinfo"),
            "Footer/contentinfo blocks should not be selected")
        assertFalse(selectedRoles.contains("banner"),
            "Banner/header blocks should not be selected")
    }

    @Test
    fun `noisy page navbar blocks score below threshold`() {
        val snapshot = loadFixture("dom_snapshot_noisy_page.json")
        
        val navBlocks = snapshot.blocks
            .map { DomBlockScorer.scoreWithBreakdown(it) }
            .filter { 
                it.block.semanticRole == "navigation" || 
                it.block.classes.any { c -> c.contains("nav", ignoreCase = true) }
            }
        
        assertTrue(navBlocks.isNotEmpty(), "Fixture should contain nav blocks")
        assertTrue(navBlocks.all { it.score < 0.25f },
            "All nav blocks should score < 0.25, got: ${navBlocks.map { it.score }}")
    }

    @Test
    fun `minimal fixture uses fallback when needed`() {
        val snapshot = loadFixture("dom_snapshot_minimal.json")
        
        val result = MenuBlockSelector.select(snapshot.blocks, ParsingMode.STRICT)
        
        // STRICT (0.45/K=4) may trigger fallback on minimal fixture
        assertTrue(result.selectedBlocks.isNotEmpty(), 
            "Should select at least one block even with minimal fixture")
    }

    @Test
    fun `full pipeline produces extractable text from clean fixture`() {
        val snapshot = loadFixture("dom_snapshot_clean_menu.json")
        
        val selection = MenuBlockSelector.select(snapshot.blocks, ParsingMode.BALANCED)
        val extractedText = BlockTextExtractor.extractFromBlocks(selection.selectedBlocks.map { it.block })
        
        assertTrue(extractedText.isNotBlank(), "Extracted text should not be blank")
        // Should contain price patterns (€XX.XX format)
        assertTrue(extractedText.contains(Regex("€\\d+[.,]\\d{2}")),
            "Extracted text should contain price patterns")
    }

    @Test
    fun `STRICT mode uses threshold 0_45 and K=4`() {
        val snapshot = loadFixture("dom_snapshot_clean_menu.json")
        
        val result = MenuBlockSelector.select(snapshot.blocks, ParsingMode.STRICT)
        
        // If no fallback, all blocks should score >= 0.45
        if (!result.usedFallback) {
            assertTrue(result.selectedBlocks.all { it.score >= 0.45f },
                "STRICT mode should only select blocks >= 0.45")
            assertTrue(result.selectedBlocks.size <= 4,
                "STRICT mode K=4 should select at most 4 blocks")
        }
        assertEquals(0.45f, result.threshold, 0.01f,
            "STRICT threshold should be 0.45")
    }

    @Test
    fun `AGGRESSIVE mode uses threshold 0_15 and K=8`() {
        val snapshot = loadFixture("dom_snapshot_noisy_page.json")
        
        val result = MenuBlockSelector.select(snapshot.blocks, ParsingMode.AGGRESSIVE)
        
        assertEquals(0.15f, result.threshold, 0.01f,
            "AGGRESSIVE threshold should be 0.15")
        assertTrue(result.selectedBlocks.size <= 8,
            "AGGRESSIVE mode K=8 should select at most 8 blocks")
    }

    @Test
    fun `fallback activates when fewer than K blocks pass threshold`() {
        // Create a snapshot where only 1-2 blocks pass STRICT threshold
        val snapshot = loadFixture("dom_snapshot_noisy_page.json")
        
        val result = MenuBlockSelector.select(snapshot.blocks, ParsingMode.STRICT)
        
        // If fallback used, verify fallback behavior
        if (result.usedFallback) {
            assertTrue(result.selectedBlocks.size >= 1,
                "Fallback should select at least top-3 or available blocks")
        }
    }

    private fun loadFixture(filename: String): DomSnapshot {
        val stream = javaClass.classLoader.getResourceAsStream(filename)
            ?: throw IllegalArgumentException("Fixture not found: $filename")
        val json = InputStreamReader(stream).readText()
        return DomSnapshotParser.parse(json)
            ?: throw IllegalStateException("Failed to parse fixture: $filename")
    }
}
