```kotlin
/** Extracts DOM snapshot with block metadata for intelligent selection.
 *  Per D-02: Returns JSON with blocks[], url, extractedAt
 *  Per D-06: Skips noise elements (nav, header, footer, role=banner/navigation) */
private val JS_EXTRACT_DOM_SNAPSHOT = """
(function() {
    try {
        // Priority 1: Check for embedded JSON data (Next.js, Nuxt)
        if (window.__NEXT_DATA__) {
            var nd = JSON.stringify(window.__NEXT_DATA__);
            if (nd.length > 500 && nd.indexOf('price') !== -1) {
                return JSON.stringify({
                    blocks: [{
                        index: 0, tag: 'SCRIPT', id: '__NEXT_DATA__',
                        classes: [], depth: 2, childCount: 0,
                        textLength: nd.length, directText: '__NEXT_DATA__',
                        subtreeText: nd.substring(0, 50000),
                        priceHits: (nd.match(/\d+[.,]\d{2}/g) || []).length,
                        foodHits: 20, visible: true,
                        rect: {x: 0, y: 0, width: 0, height: 0},
                        semanticRole: 'data', schemaType: 'Menu',
                        xpath: '/html/head/script[@id="__NEXT_DATA__"]'
                    }],
                    url: window.location.href,
                    extractedAt: Date.now()
                });
            }
        }

        // Food words for hit counting (50 Italian + 50 English per D-06)
        var foodWords = [
            // Italian
            'pasta','pizza','risotto','gnocchi','lasagna','ravioli','tortellini',
            'antipasto','antipasti','bruschetta','carpaccio','prosciutto','salame',
            'mozzarella','parmigiano','pecorino','gorgonzola','burrata',
            'pollo','manzo','vitello','maiale','agnello','pesce','gamberi',
            'salmone','tonno','branzino','orata','calamari','polpo','cozze',
            'verdure','insalata','pomodoro','melanzane','zucchine','funghi',
            'tartufo','basilico','origano','olio','aceto','vino','dolce',
            'tiramisu','panna','gelato','cannoli','biscotti',
            // English
            'appetizer','starter','main','entree','dessert','salad','soup',
            'burger','steak','chicken','beef','pork','lamb','fish','salmon',
            'shrimp','lobster','crab','oyster','tuna','cod','seafood',
            'vegetable','mushroom','potato','rice','noodle','bread','cheese',
            'wine','beer','cocktail','coffee','tea','juice','water',
            'grilled','fried','baked','roasted','steamed','sauteed','fresh',
            'organic','vegan','vegetarian','gluten','sauce','dressing'
        ];
        var foodRegex = new RegExp('\\\\b(' + foodWords.join('|') + ')\\\\b', 'gi');
        
        // Price regex (handles €12.50, $9.99, 15,00€, etc.)
        var priceRegex = /(?:€|\$|EUR|USD)?\s*\d{1,3}[.,]\d{2}\s*(?:€|\$|EUR|USD)?/g;
        
        // Noise selectors per D-06
        var noiseSelectors = [
            'nav', 'header', 'footer', 'aside',
            '[role="banner"]', '[role="navigation"]', '[role="contentinfo"]',
            'script', 'style', 'noscript', 'iframe', 'svg'
        ];
        
        var blocks = [];
        var index = 0;
        
        // TreeWalker for all elements
        var walker = document.createTreeWalker(
            document.body || document.documentElement,
            NodeFilter.SHOW_ELEMENT,
            null,
            false
        );
        
        var el = walker.currentNode;
        while (el) {
            try {
                var tag = (el.tagName || '').toUpperCase();
                
                // Skip noise elements per D-06
                var isNoise = noiseSelectors.some(function(sel) {
                    try {
                        if (sel.startsWith('[')) {
                            return el.matches && el.matches(sel);
                        }
                        return tag.toLowerCase() === sel;
                    } catch(e) { return false; }
                });
                
                if (!isNoise) {
                    // Check visibility
                    var visible = true;
                    try {
                        var cs = window.getComputedStyle(el);
                        if (cs.display === 'none' || cs.visibility === 'hidden') {
                            visible = false;
                        }
                    } catch(e) {}
                    
                    if (visible) {
                        // Get text content
                        var subtreeText = (el.innerText || el.textContent || '').trim();
                        var directText = '';
                        for (var i = 0; i < el.childNodes.length; i++) {
                            if (el.childNodes[i].nodeType === 3) {
                                directText += el.childNodes[i].textContent || '';
                            }
                        }
                        directText = directText.replace(/\s+/g, ' ').trim();
                        
                        var textLength = subtreeText.length;
                        
                        // Count hits
                        var priceHits = (subtreeText.match(priceRegex) || []).length;
                        var foodHits = (subtreeText.match(foodRegex) || []).length;
                        
                        // Emit block if meets criteria per D-06
                        if (textLength > 100 || priceHits > 0) {
                            // Get bounding rect
                            var rect = {x: 0, y: 0, width: 0, height: 0};
                            try {
                                var r = el.getBoundingClientRect();
                                rect = {x: r.left, y: r.top, width: r.width, height: r.height};
                            } catch(e) {}
                            
                            // Calculate depth
                            var depth = 0;
                            var parent = el.parentElement;
                            while (parent) { depth++; parent = parent.parentElement; }
                            
                            // Get semantic signals
                            var role = el.getAttribute('role') || null;
                            var schemaType = el.getAttribute('itemtype') || null;
                            if (schemaType && schemaType.indexOf('Menu') !== -1) {
                                schemaType = 'Menu';
                            }
                            
                            // Check for data-menu attribute
                            if (el.hasAttribute('data-menu')) {
                                role = 'data-menu';
                            }
                            
                            // Build xpath
                            var xpath = '';
                            try {
                                var path = [];
                                var current = el;
                                while (current && current.nodeType === 1) {
                                    var name = current.tagName.toLowerCase();
                                    var idx = 1;
                                    var sib = current.previousElementSibling;
                                    while (sib) {
                                        if (sib.tagName === current.tagName) idx++;
                                        sib = sib.previousElementSibling;
                                    }
                                    path.unshift(name + '[' + idx + ']');
                                    current = current.parentElement;
                                }
                                xpath = '/' + path.join('/');
                            } catch(e) { xpath = '/unknown'; }
                            
                            blocks.push({
                                index: index++,
                                tag: tag,
                                id: el.id || null,
                                classes: el.className ? el.className.split(/\s+/).filter(function(c) { return c; }) : [],
                                depth: depth,
                                childCount: el.childElementCount || 0,
                                textLength: textLength,
                                directText: directText.substring(0, 200),
                                subtreeText: subtreeText.substring(0, 10000),
                                priceHits: priceHits,
                                foodHits: foodHits,
                                visible: visible,
                                rect: rect,
                                semanticRole: role,
                                schemaType: schemaType,
                                xpath: xpath
                            });
                        }
                    }
                }
            } catch(e) {}
            
            el = walker.nextNode();
        }
        
        return JSON.stringify({
            blocks: blocks,
            url: window.location.href,
            extractedAt: Date.now()
        });
        
    } catch(e) {
        return JSON.stringify({
            blocks: [],
            url: window.location.href,
            extractedAt: Date.now(),
            error: e.message
        });
    }
})()
""".trimIndent()
```

```kotlin
private fun extractAndComplete(
    view: WebView,
    deferred: CompletableDeferred<WebExtractResult>,
    capturedApiJson: MutableList<String>,
    mode: com.example.ingredient.parser.ParsingMode = com.example.ingredient.parser.ParsingMode.BALANCED
) {
    // Try DOM snapshot extraction first
    view.evaluateJavascript(JS_EXTRACT_DOM_SNAPSHOT) { rawSnapshot ->
        val snapshot = com.example.ingredient.parser.dom.DomSnapshotParser.parse(rawSnapshot ?: "")
        
        if (snapshot != null && snapshot.blocks.isNotEmpty()) {
            // Success: use DOM-aware extraction
            val extraction = com.example.ingredient.parser.dom.BlockTextExtractor
                .extractFromSnapshot(snapshot, mode)
            
            Log.d(EXTRACTOR_TAG, "DOM extraction: ${snapshot.blocks.size} blocks, " +
                "${extraction.selection.selectedCount} selected, ${extraction.textLength} chars")
            
            if (!deferred.isCompleted) {
                deferred.complete(WebExtractResult(
                    visibleText = extraction.text,
                    capturedApiJson = capturedApiJson.toList(),
                    domSnapshot = snapshot
                ))
            }
        } else {
            // Fallback to legacy JS_EXTRACT_TEXT
            Log.d(EXTRACTOR_TAG, "DOM snapshot failed, falling back to JS_EXTRACT_TEXT")
            view.evaluateJavascript(JS_EXTRACT_TEXT) { raw ->
                val text = raw
                    ?.removePrefix("\"")
                    ?.removeSuffix("\"")
                    ?.replace("\\n", "\n")
                    ?.replace("\\t", "\t")
                    ?.replace("\\\"", "\"")
                    ?.replace("\\/", "/")
                    ?: ""
                if (!deferred.isCompleted) {
                    deferred.complete(WebExtractResult(text, capturedApiJson.toList(), null))
                }
            }
        }
    }
}
```

```kotlin
/**
 * Log block selection trace with per-block scores.
 * Per D-09: ParseLogger emits block selection trace with scores.

fun logBlockSelection(
    selection: com.example.ingredient.parser.dom.SelectionResult,
    groups: List<com.example.ingredient.parser.dom.MergedBlockGroup>
) {
    val sb = StringBuilder()
    sb.appendLine("══ BLOCK SELECTION ══")
    sb.appendLine("  total_blocks:    ${selection.totalBlocks}")
    sb.appendLine("  selected:        ${selection.selectedCount}")
    sb.appendLine("  threshold:       ${"%.3f".format(selection.threshold)}")
    sb.appendLine("  used_fallback:   ${selection.usedFallback}")
    sb.appendLine("  merged_groups:   ${groups.size}")
    
    sb.appendLine("  ── Selected Blocks ──")
    selection.selectedBlocks.forEachIndexed { i, scored ->
        val block = scored.block
        sb.appendLine("    [$i] score=${"%.3f".format(scored.score)} " +
            "tag=${block.tag} depth=${block.depth} " +
            "prices=${block.priceHits} foods=${block.foodHits} " +
            "text=${block.directText.take(40)}...")
        
        // Log breakdown
        sb.appendLine("        breakdown: " +
            "price=${"%.2f".format(scored.breakdown.priceDensity)} " +
            "food=${"%.2f".format(scored.breakdown.foodDensity)} " +
            "semantic=${"%.2f".format(scored.breakdown.semanticRole)} " +
            "class=${"%.2f".format(scored.breakdown.className)} " +
            "pos=${"%.2f".format(scored.breakdown.position)}")
    }
    
    if (groups.any { it.size > 1 }) {
        sb.appendLine("  ── Merged Groups ──")
        groups.filter { it.size > 1 }.forEachIndexed { i, group ->
            sb.appendLine("    Group $i: ${group.size} blocks, reason=${group.mergeReason}")
        }
    }
    
    sb.appendLine("═══════════════════════")
    Log.d(TAG, sb.toString())
}
```
