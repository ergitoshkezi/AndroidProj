```kotlin
package com.example.ingredient.parser.dom

/**
 * Classification result for a DOM block.
 */
enum class BlockClassification {
    /** High confidence menu content (score >= threshold) */
    MENU,
    /** Definite noise - nav, footer, etc. (score < low_threshold OR noise signals) */
    NOISE,
    /** Uncertain - may or may not be menu (low_threshold <= score < threshold) */
    AMBIGUOUS
}

/**
 * Classifies DOM blocks as MENU, NOISE, or AMBIGUOUS based on score and signals.
 * Works in conjunction with MenuBlockSelector for final selection.
 */
object MenuBlockClassifier {
    
    // Noise indicators (per D-06: skip nav, header, footer, role=banner/navigation)
    private val NOISE_TAGS = setOf("nav", "header", "footer")
    private val NOISE_ROLES = setOf("banner", "navigation", "contentinfo")
    
    /**
     * Classify a single block based on its score and characteristics.
     * @param scoredBlock Block with pre-computed score
     * @param threshold Score threshold for MENU classification (varies by ParsingMode)
     * @return BlockClassification
     */
    fun classify(scoredBlock: ScoredBlock, threshold: Float): BlockClassification {
        val block = scoredBlock.block
        
        // Hard noise rules (per D-06) - these override score
        if (isDefiniteNoise(block)) {
            return BlockClassification.NOISE
        }
        
        // Score-based classification
        return when {
            scoredBlock.score >= threshold -> BlockClassification.MENU
            scoredBlock.score >= threshold * 0.6f -> BlockClassification.AMBIGUOUS
            else -> BlockClassification.NOISE
        }
    }
    
    /**
     * Classify a list of scored blocks.
     */
    fun classifyAll(
        scoredBlocks: List<ScoredBlock>,
        threshold: Float
    ): List<ClassifiedBlock> {
        return scoredBlocks.map { scored ->
            ClassifiedBlock(
                scoredBlock = scored,
                classification = classify(scored, threshold)
            )
        }
    }
    
    /**
     * Check if block is definite noise based on tag/role (per D-06).
     */
    private fun isDefiniteNoise(block: DomBlock): Boolean {
        // Check tag
        if (block.tag.lowercase() in NOISE_TAGS) return true
        
        // Check semantic role
        if (block.semanticRole?.lowercase() in NOISE_ROLES) return true
        
        // Check for hidden blocks (shouldn't be here, but defensive)
        if (!block.visible) return true
        
        return false
    }
}

/**
 * A block with score and classification.
 */
data class ClassifiedBlock(
    val scoredBlock: ScoredBlock,
    val classification: BlockClassification
) {
    val block: DomBlock get() = scoredBlock.block
    val score: Float get() = scoredBlock.score
}
```

```kotlin
package com.example.ingredient.parser.dom

import com.example.ingredient.parser.ast.ParseMetadata

/**
 * Selects top-K menu blocks based on ParsingMode configuration.
 * 
 * Per D-04 (Recall-first):
 * - Default threshold: 0.25, K=8
 * - Fallback: if 0 blocks pass → lower to 0.15, accept top-3
 * 
 * Per D-05 (ParsingMode integration):
 * - STRICT: threshold >= 0.45, K=4
 * - BALANCED: threshold >= 0.25, K=6
 * - AGGRESSIVE: threshold >= 0.15, K=8
 */
object MenuBlockSelector {
    
    /**
     * ParsingMode-specific selection parameters.
     */
    data class SelectionParams(
        val threshold: Float,
        val maxBlocks: Int,
        val fallbackThreshold: Float,
        val fallbackMaxBlocks: Int
    )
    
    // Per D-05 (LOCKED)
    private val STRICT_PARAMS = SelectionParams(
        threshold = 0.45f,
        maxBlocks = 4,
        fallbackThreshold = 0.25f,
        fallbackMaxBlocks = 2
    )
    
    private val BALANCED_PARAMS = SelectionParams(
        threshold = 0.25f,
        maxBlocks = 6,
        fallbackThreshold = 0.15f,
        fallbackMaxBlocks = 3
    )
    
    private val AGGRESSIVE_PARAMS = SelectionParams(
        threshold = 0.15f,
        maxBlocks = 8,
        fallbackThreshold = 0.10f,
        fallbackMaxBlocks = 3
    )
    
    /**
     * Select top menu blocks from snapshot using ParsingMode configuration.
     * @param blocks All blocks from DomSnapshot
     * @param mode ParsingMode determining thresholds (STRICT/BALANCED/AGGRESSIVE)
     * @return Selected blocks sorted by document order (index)
     */
    fun select(
        blocks: List<DomBlock>,
        mode: com.example.ingredient.parser.ParsingMode
    ): SelectionResult {
        val params = getParams(mode)
        
        // Score all blocks
        val scored = blocks.map { DomBlockScorer.scoreWithBreakdown(it) }
        
        // Filter by threshold and take top-K by score
        var selected = scored
            .filter { it.score >= params.threshold }
            .sortedByDescending { it.score }
            .take(params.maxBlocks)
        
        // Fallback per D-04: if 0 blocks pass, lower threshold and take top-3
        if (selected.isEmpty()) {
            selected = scored
                .filter { it.score >= params.fallbackThreshold }
                .sortedByDescending { it.score }
                .take(params.fallbackMaxBlocks)
        }
        
        // Sort by document order for text extraction (index field)
        val orderedBlocks = selected.sortedBy { it.block.index }
        
        return SelectionResult(
            selectedBlocks = orderedBlocks,
            totalBlocks = blocks.size,
            threshold = if (selected.isEmpty() || scored.any { it.score >= params.threshold }) 
                params.threshold else params.fallbackThreshold,
            usedFallback = selected.isNotEmpty() && 
                scored.none { it.score >= params.threshold }
        )
    }
    
    /**
     * Get selection parameters for a ParsingMode.
     */
    fun getParams(mode: com.example.ingredient.parser.ParsingMode): SelectionParams {
        return when (mode) {
            com.example.ingredient.parser.ParsingMode.STRICT -> STRICT_PARAMS
            com.example.ingredient.parser.ParsingMode.BALANCED -> BALANCED_PARAMS
            com.example.ingredient.parser.ParsingMode.AGGRESSIVE -> AGGRESSIVE_PARAMS
        }
    }
}

/**
 * Result of block selection with metadata for logging.
 */
data class SelectionResult(
    val selectedBlocks: List<ScoredBlock>,
    val totalBlocks: Int,
    val threshold: Float,
    val usedFallback: Boolean
) {
    val selectedCount: Int get() = selectedBlocks.size
}
```

```kotlin
package com.example.ingredient.parser.dom

/**
 * Merges adjacent candidate blocks to preserve document continuity.
 * 
 * Per D-07 (LOCKED):
 * - Merge if two candidate blocks are siblings OR depth gap <= 2
 * - Preserves document order in merged text
 */
object BlockMerger {
    
    /**
     * Merge adjacent selected blocks based on D-07 rules.
     * @param blocks Selected blocks sorted by index (document order)
     * @return Merged block groups
     */
    fun merge(blocks: List<ScoredBlock>): List<MergedBlockGroup> {
        if (blocks.isEmpty()) return emptyList()
        if (blocks.size == 1) {
            return listOf(MergedBlockGroup(
                blocks = blocks,
                mergeReason = null
            ))
        }
        
        val groups = mutableListOf<MergedBlockGroup>()
        var currentGroup = mutableListOf(blocks.first())
        
        for (i in 1 until blocks.size) {
            val prev = blocks[i - 1]
            val curr = blocks[i]
            
            val mergeReason = shouldMerge(prev.block, curr.block)
            
            if (mergeReason != null) {
                // Merge into current group
                currentGroup.add(curr)
            } else {
                // Start new group
                groups.add(MergedBlockGroup(
                    blocks = currentGroup.toList(),
                    mergeReason = if (currentGroup.size > 1) getMergeReason(currentGroup) else null
                ))
                currentGroup = mutableListOf(curr)
            }
        }
        
        // Add final group
        groups.add(MergedBlockGroup(
            blocks = currentGroup.toList(),
            mergeReason = if (currentGroup.size > 1) getMergeReason(currentGroup) else null
        ))
        
        return groups
    }
    
    /**
     * Determine if two blocks should be merged.
     * Per D-07: siblings OR depth gap <= 2
     * @return Merge reason string or null if should not merge
     */
    private fun shouldMerge(a: DomBlock, b: DomBlock): String? {
        // Check depth gap
        val depthGap = kotlin.math.abs(a.depth - b.depth)
        if (depthGap <= 2) {
            return "depth_gap=$depthGap"
        }
        
        // Check sibling relationship via xpath
        if (areSiblings(a.xpath, b.xpath)) {
            return "siblings"
        }
        
        // Check consecutive index (adjacent in document order)
        if (b.index - a.index == 1) {
            return "consecutive"
        }
        
        return null
    }
    
    /**
     * Check if two xpaths indicate sibling elements.
     * Siblings have same parent path, differ only in last index.
     * Example: /html/body/main/section[2] and /html/body/main/section[3] are siblings
     */
    private fun areSiblings(xpathA: String, xpathB: String): Boolean {
        val parentA = getParentXpath(xpathA)
        val parentB = getParentXpath(xpathB)
        return parentA != null && parentA == parentB
    }
    
    /**
     * Extract parent xpath (everything before last path segment).
     */
    private fun getParentXpath(xpath: String): String? {
        val lastSlash = xpath.lastIndexOf('/')
        return if (lastSlash > 0) xpath.substring(0, lastSlash) else null
    }
    
    /**
     * Get overall merge reason for a group.
     */
    private fun getMergeReason(blocks: List<ScoredBlock>): String {
        val reasons = mutableSetOf<String>()
        for (i in 1 until blocks.size) {
            shouldMerge(blocks[i-1].block, blocks[i].block)?.let { reasons.add(it) }
        }
        return reasons.joinToString(", ")
    }
}

/**
 * A group of merged blocks.
 */
data class MergedBlockGroup(
    val blocks: List<ScoredBlock>,
    val mergeReason: String?
) {
    val size: Int get() = blocks.size
    val totalScore: Float get() = blocks.sumOf { it.score.toDouble() }.toFloat()
    val averageScore: Float get() = if (blocks.isEmpty()) 0f else totalScore / blocks.size
}
```

```kotlin
package com.example.ingredient.parser.dom

/**
 * Extracts ordered text from selected DOM blocks for MenuParserPipeline input.
 * Preserves document order and adds category markers for headings.
 */
object BlockTextExtractor {
    
    /**
     * Extract text from merged block groups.
     * @param groups Merged block groups from BlockMerger
     * @return Concatenated text preserving document order
     */
    fun extract(groups: List<MergedBlockGroup>): String {
        if (groups.isEmpty()) return ""
        
        return groups.flatMap { group ->
            group.blocks.map { scored ->
                formatBlockText(scored.block)
            }
        }.joinToString("\n\n")
    }
    
    /**
     * Extract text from selected blocks (without merging).
     * @param blocks Selected blocks from MenuBlockSelector
     * @return Concatenated text preserving document order
     */
    fun extractFromBlocks(blocks: List<ScoredBlock>): String {
        if (blocks.isEmpty()) return ""
        
        // Sort by index (document order) just in case
        return blocks
            .sortedBy { it.block.index }
            .map { formatBlockText(it.block) }
            .joinToString("\n\n")
    }
    
    /**
     * Extract text from raw DomBlocks.
     */
    fun extractFromDomBlocks(blocks: List<DomBlock>): String {
        if (blocks.isEmpty()) return ""
        
        return blocks
            .sortedBy { it.index }
            .map { formatBlockText(it) }
            .joinToString("\n\n")
    }
    
    /**
     * Format a single block's text with optional category marker.
     */
    private fun formatBlockText(block: DomBlock): String {
        val text = block.subtreeText.trim()
        
        // Add category marker for semantic menu sections
        val prefix = when {
            block.schemaType?.equals("Menu", ignoreCase = true) == true -> 
                "=== MENU: ${block.directText.take(50)} ===\n"
            block.semanticRole?.equals("menu", ignoreCase = true) == true ->
                "=== CATEGORIA: ${block.directText.take(50)} ===\n"
            isHeadingTag(block.tag) && block.directText.length <= 60 ->
                "=== SEZIONE: ${block.directText} ===\n"
            else -> ""
        }
        
        return prefix + text
    }
    
    /**
     * Check if tag is a heading that should get a category marker.
     */
    private fun isHeadingTag(tag: String): Boolean {
        return tag.uppercase() in setOf("H1", "H2", "H3", "H4", "SECTION", "ARTICLE")
    }
    
    /**
     * Full extraction pipeline: select → merge → extract.
     * Convenience method combining all stages.
     */
    fun extractFromSnapshot(
        snapshot: DomSnapshot,
        mode: com.example.ingredient.parser.ParsingMode
    ): ExtractionResult {
        // Select
        val selectionResult = MenuBlockSelector.select(snapshot.blocks, mode)
        
        // Merge
        val groups = BlockMerger.merge(selectionResult.selectedBlocks)
        
        // Extract
        val text = extract(groups)
        
        return ExtractionResult(
            text = text,
            selection = selectionResult,
            groups = groups
        )
    }
}

/**
 * Complete extraction result with metadata for logging.
 */
data class ExtractionResult(
    val text: String,
    val selection: SelectionResult,
    val groups: List<MergedBlockGroup>
) {
    val textLength: Int get() = text.length
    val blockCount: Int get() = selection.selectedCount
    val groupCount: Int get() = groups.size
}
```
