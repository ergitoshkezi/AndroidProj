```kotlin
package com.example.ingredient.parser.dom

import com.google.gson.Gson
import com.google.gson.JsonSyntaxException

/**
 * Parses JSON string from JS_EXTRACT_DOM_SNAPSHOT into DomSnapshot.
 * Uses Gson (already in project dependencies).
 */
object DomSnapshotParser {
    
    private val gson = Gson()
    
    /**
     * Parse JSON string to DomSnapshot.
     * @param json Raw JSON string from JavaScript evaluateJavascript callback
     * @return DomSnapshot or null if parsing fails
     */
    fun parse(json: String): DomSnapshot? {
        return try {
            // JavaScript evaluateJavascript returns escaped string with quotes
            val cleanJson = json
                .removePrefix("\"")
                .removeSuffix("\"")
                .replace("\\\"", "\"")
                .replace("\\n", "\n")
                .replace("\\/", "/")
            
            gson.fromJson(cleanJson, DomSnapshot::class.java)
        } catch (e: JsonSyntaxException) {
            null
        } catch (e: Exception) {
            null
        }
    }
    
    /**
     * Parse with detailed error for debugging.
     * @return Pair of DomSnapshot (or null) and error message (or null)
     */
    fun parseWithError(json: String): Pair<DomSnapshot?, String?> {
        return try {
            val cleanJson = json
                .removePrefix("\"")
                .removeSuffix("\"")
                .replace("\\\"", "\"")
                .replace("\\n", "\n")
                .replace("\\/", "/")
            
            Pair(gson.fromJson(cleanJson, DomSnapshot::class.java), null)
        } catch (e: JsonSyntaxException) {
            Pair(null, "JSON syntax error: ${e.message}")
        } catch (e: Exception) {
            Pair(null, "Parse error: ${e.message}")
        }
    }
}
```

```kotlin
package com.example.ingredient.parser.dom

/**
 * Scores DOM blocks for menu-relevance using deterministic 5-signal formula.
 * Per D-03 (LOCKED):
 *   score = 0.30 × priceDensity
 *         + 0.25 × foodWordDensity  
 *         + 0.20 × semanticRoleScore
 *         + 0.15 × classNameScore
 *         + 0.10 × positionScore
 */
object DomBlockScorer {
    
    // Weights per D-03 (LOCKED - do not change)
    private const val W_PRICE_DENSITY = 0.30f
    private const val W_FOOD_DENSITY = 0.25f
    private const val W_SEMANTIC_ROLE = 0.20f
    private const val W_CLASS_NAME = 0.15f
    private const val W_POSITION = 0.10f
    
    // Regex for classNameScore (menu-relevant class/id patterns)
    private val MENU_CLASS_REGEX = Regex(
        "menu|piatto|dish|categ|food|item|carta",
        RegexOption.IGNORE_CASE
    )
    
    /**
     * Score a single block for menu-relevance.
     * @return Float in range [0.0, 1.0]
     */
    fun score(block: DomBlock): Float {
        val priceDensity = calcPriceDensity(block)
        val foodDensity = calcFoodDensity(block)
        val semanticRole = calcSemanticRoleScore(block)
        val className = calcClassNameScore(block)
        val position = calcPositionScore(block)
        
        return (W_PRICE_DENSITY * priceDensity +
                W_FOOD_DENSITY * foodDensity +
                W_SEMANTIC_ROLE * semanticRole +
                W_CLASS_NAME * className +
                W_POSITION * position).coerceIn(0f, 1f)
    }
    
    /**
     * Score with breakdown for observability (ParseLogger integration per D-09).
     */
    fun scoreWithBreakdown(block: DomBlock): ScoredBlock {
        val priceDensity = calcPriceDensity(block)
        val foodDensity = calcFoodDensity(block)
        val semanticRole = calcSemanticRoleScore(block)
        val className = calcClassNameScore(block)
        val position = calcPositionScore(block)
        
        val total = (W_PRICE_DENSITY * priceDensity +
                     W_FOOD_DENSITY * foodDensity +
                     W_SEMANTIC_ROLE * semanticRole +
                     W_CLASS_NAME * className +
                     W_POSITION * position).coerceIn(0f, 1f)
        
        return ScoredBlock(
            block = block,
            score = total,
            breakdown = ScoreBreakdown(
                priceDensity = priceDensity,
                foodDensity = foodDensity,
                semanticRole = semanticRole,
                className = className,
                position = position
            )
        )
    }
    
    // Per D-03: priceDensity = priceHits / (textLength / 1000).coerceAtLeast(1f)
    private fun calcPriceDensity(block: DomBlock): Float {
        val normalizedLength = (block.textLength / 1000f).coerceAtLeast(1f)
        return (block.priceHits / normalizedLength).coerceAtMost(1f)
    }
    
    // Per D-03: foodWordDensity = foodHits / (textLength / 500).coerceAtLeast(1f)
    private fun calcFoodDensity(block: DomBlock): Float {
        val normalizedLength = (block.textLength / 500f).coerceAtLeast(1f)
        return (block.foodHits / normalizedLength).coerceAtMost(1f)
    }
    
    // Per D-03: 1.0 if role=menu/schemaType=Menu, 0.7 if data-menu attr, 0.3 if aria-label contains "menu"
    private fun calcSemanticRoleScore(block: DomBlock): Float {
        // Check schemaType first (highest signal)
        if (block.schemaType?.equals("Menu", ignoreCase = true) == true) return 1.0f
        
        // Check semanticRole
        when (block.semanticRole?.lowercase()) {
            "menu" -> return 1.0f
            "data-menu" -> return 0.7f  // JS will set this for data-menu attribute
        }
        
        // Check if any class suggests menu (aria-label check happens in classNameScore)
        if (block.classes.any { it.contains("menu", ignoreCase = true) }) {
            return 0.3f
        }
        
        return 0f
    }
    
    // Per D-03: regex `menu|piatto|dish|categ|food|item|carta` on classes+id
    private fun calcClassNameScore(block: DomBlock): Float {
        val combinedText = buildString {
            block.id?.let { append(it).append(" ") }
            append(block.classes.joinToString(" "))
        }
        
        return if (MENU_CLASS_REGEX.containsMatchIn(combinedText)) 1.0f else 0f
    }
    
    // Per D-03: 1.0 if rect.y > 200px AND depth 3-6, 0.5 otherwise
    private fun calcPositionScore(block: DomBlock): Float {
        val goodVerticalPosition = block.rect.y > 200f
        val goodDepth = block.depth in 3..6
        
        return if (goodVerticalPosition && goodDepth) 1.0f else 0.5f
    }
}

/**
 * A block with its computed score and breakdown.
 */
data class ScoredBlock(
    val block: DomBlock,
    val score: Float,
    val breakdown: ScoreBreakdown
)

/**
 * Individual signal contributions for observability.
 */
data class ScoreBreakdown(
    val priceDensity: Float,
    val foodDensity: Float,
    val semanticRole: Float,
    val className: Float,
    val position: Float
)
```
