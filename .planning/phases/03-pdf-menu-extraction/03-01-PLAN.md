```kotlin
package com.example.ingredient.parser.pdf

import android.util.Log
import com.tom_roush.pdfbox.android.PDFBoxResourceLoader
import com.tom_roush.pdfbox.pdmodel.PDDocument
import com.tom_roush.pdfbox.text.PDFTextStripper
import java.io.File

object PdfTextLayerExtractor {
    private const val TAG = "PdfTextLayerExtractor"

    /**
     * Extracts text from each page of a PDF file using PDFBox-Android.
     * Returns a list of strings, one per page. Empty pages return empty strings.
     * Returns empty list if PDF cannot be loaded or has no pages.
     */
    fun extract(pdfFile: File, context: android.content.Context): List<String> {
        return try {
            // Initialize PDFBox resources (required once per app lifecycle, safe to call multiple times)
            PDFBoxResourceLoader.init(context)

            PDDocument.load(pdfFile).use { document ->
                val pageCount = document.numberOfPages
                if (pageCount == 0) {
                    Log.w(TAG, "PDF has no pages")
                    return emptyList()
                }

                val stripper = PDFTextStripper()
                val pages = mutableListOf<String>()

                for (i in 1..pageCount) {
                    stripper.startPage = i
                    stripper.endPage = i
                    val pageText = stripper.getText(document) ?: ""
                    pages.add(pageText)
                    Log.d(TAG, "Page $i: ${pageText.length} chars")
                }

                Log.d(TAG, "Extracted $pageCount pages, total ${pages.sumOf { it.length }} chars")
                pages
            }
        } catch (e: Exception) {
            Log.e(TAG, "Failed to extract text layer: ${e.message}", e)
            emptyList()
        }
    }
}
```

Key points:
- Uses `PDDocument.load()` with Kotlin `use` for auto-close
- Extracts page-by-page using `PDFTextStripper` with startPage/endPage
- Calls `PDFBoxResourceLoader.init()` (required for Android, idempotent)
- Returns empty list on failure (allows graceful fallback to OCR)

```kotlin
package com.example.ingredient.parser.pdf

import android.util.Log
import com.example.ingredient.parser.ast.ParsingMode

object PdfQualityChecker {
    private const val TAG = "PdfQualityChecker"

    // D-02 thresholds
    private const val EXPECTED_CHARS_PER_PAGE = 500
    private const val MIN_LINE_LENGTH = 10
    private const val MAX_LINE_LENGTH = 120
    private const val MIN_TOKEN_LENGTH = 3

    // Price pattern: €X.XX, $X.XX, X,XX€, X.XX (with optional currency)
    private val PRICE_PATTERN = Regex(
        """(?:€|£|\$)?\s*\d{1,3}(?:[.,]\d{2})?\s*(?:€|£|\$)?"""
    )

    /**
     * Scores the quality of extracted PDF text pages.
     * Returns a value between 0.0 and 1.0.
     *
     * Formula (D-02):
     * textScore = 0.40 × charDensity + 0.30 × wordRecognition + 0.20 × pricePresence + 0.10 × lineStructure
     */
    fun score(pages: List<String>): Float {
        if (pages.isEmpty()) return 0f

        val combinedText = pages.joinToString("\n")
        if (combinedText.isBlank()) return 0f

        val charDensity = calculateCharDensity(combinedText, pages.size)
        val wordRecognition = calculateWordRecognition(combinedText)
        val pricePresence = calculatePricePresence(combinedText)
        val lineStructure = calculateLineStructure(combinedText)

        val score = (0.40f * charDensity +
                     0.30f * wordRecognition +
                     0.20f * pricePresence +
                     0.10f * lineStructure).coerceIn(0f, 1f)

        Log.d(TAG, "Quality score: %.3f (charDensity=%.2f, wordRecog=%.2f, price=%.2f, structure=%.2f)"
            .format(score, charDensity, wordRecognition, pricePresence, lineStructure))

        return score
    }

    /**
     * Determines if the text layer is usable based on score and parsing mode.
     * Per D-03:
     * - STRICT: >= 0.70
     * - BALANCED: >= 0.50
     * - AGGRESSIVE: always true (try text first)
     */
    fun isUsable(score: Float, mode: ParsingMode): Boolean {
        val threshold = when (mode) {
            ParsingMode.STRICT -> 0.70f
            ParsingMode.BALANCED -> 0.50f
            ParsingMode.AGGRESSIVE -> 0f // Always try text first
        }
        val usable = score >= threshold
        Log.d(TAG, "isUsable: score=%.3f, mode=$mode, threshold=%.2f → $usable"
            .format(score, threshold))
        return usable
    }

    // charDensity: extracted chars / (pageCount × 500)
    private fun calculateCharDensity(text: String, pageCount: Int): Float {
        val charCount = text.count { !it.isWhitespace() }
        val expectedChars = pageCount * EXPECTED_CHARS_PER_PAGE
        return (charCount.toFloat() / expectedChars).coerceIn(0f, 1f)
    }

    // wordRecognition: % tokens that match [\\w]{3,} / totalTokens
    private fun calculateWordRecognition(text: String): Float {
        val tokens = text.split(Regex("\\s+")).filter { it.isNotBlank() }
        if (tokens.isEmpty()) return 0f

        val validTokens = tokens.count { token ->
            token.length >= MIN_TOKEN_LENGTH &&
            token.any { it.isLetter() } &&
            !token.all { !it.isLetterOrDigit() }
        }
        return validTokens.toFloat() / tokens.size
    }

    // pricePresence: min(pricePatternCount / 3, 1.0)
    private fun calculatePricePresence(text: String): Float {
        val priceCount = PRICE_PATTERN.findAll(text).count()
        return (priceCount / 3f).coerceIn(0f, 1f)
    }

    // lineStructure: % lines with 10-120 chars length
    private fun calculateLineStructure(text: String): Float {
        val lines = text.lines().filter { it.isNotBlank() }
        if (lines.isEmpty()) return 0f

        val wellStructuredLines = lines.count { line ->
            val trimmed = line.trim()
            trimmed.length in MIN_LINE_LENGTH..MAX_LINE_LENGTH
        }
        return wellStructuredLines.toFloat() / lines.size
    }
}
```

- Mode-based routing per D-03
- Price pattern catches €, $, £ with common formats
- All component scores clamped to [0, 1]

```kotlin
package com.example.ingredient.parser.pdf

import android.util.Log

object PdfTextCleaner {
    private const val TAG = "PdfTextCleaner"

    // Unicode ligature replacements (D-05)
    private val LIGATURES = mapOf(
        '\uFB01' to "fi",  // ﬁ ligature
        '\uFB02' to "fl",  // ﬂ ligature
        '\uFB03' to "ffi", // ﬃ ligature
        '\uFB04' to "ffl", // ﬄ ligature
        '\uFB00' to "ff",  // ﬀ ligature
        '\uFB05' to "st",  // ﬅ long s t ligature
        '\uFB06' to "st",  // ﬆ st ligature
    )

    // Soft hyphen and other invisible chars
    private const val SOFT_HYPHEN = '\u00AD'
    private const val ZERO_WIDTH_SPACE = '\u200B'
    private const val ZERO_WIDTH_NON_JOINER = '\u200C'
    private const val ZERO_WIDTH_JOINER = '\u200D'

    /**
     * Cleans extracted PDF text by fixing common artifacts:
     * - Ligatures: fi, fl, ffi, etc. → ASCII equivalents
     * - Soft hyphens: word-\nnext → wordnext
     * - Extra whitespace between chars (char-by-char extraction)
     * - Page headers/footers (repeated patterns)
     *
     * Returns a single cleaned string from all pages.
     */
    fun clean(pages: List<String>): String {
        if (pages.isEmpty()) return ""

        val cleaned = pages.mapIndexed { index, page ->
            val cleanedPage = cleanPage(page)
            if (cleanedPage.isNotBlank()) {
                "\n\n=== PAGINA ${index + 1} ===\n$cleanedPage"
            } else ""
        }.joinToString("")

        Log.d(TAG, "Cleaned ${pages.size} pages → ${cleaned.length} chars")
        return cleaned.trim()
    }

    private fun cleanPage(text: String): String {
        var result = text

        // 1. Replace ligatures
        LIGATURES.forEach { (ligature, replacement) ->
            result = result.replace(ligature.toString(), replacement)
        }

        // 2. Remove zero-width characters
        result = result
            .replace(ZERO_WIDTH_SPACE.toString(), "")
            .replace(ZERO_WIDTH_NON_JOINER.toString(), "")
            .replace(ZERO_WIDTH_JOINER.toString(), "")

        // 3. Fix soft hyphenation: "word-\n  next" → "wordnext"
        result = result.replace(Regex("""(\w)$SOFT_HYPHEN?\s*-\s*\n\s*(\w)""")) { match ->
            match.groupValues[1] + match.groupValues[2]
        }

        // 4. Also fix visible hyphen at end of line followed by word continuation
        result = result.replace(Regex("""(\w)-\n\s*([a-z])""")) { match ->
            match.groupValues[1] + match.groupValues[2]
        }

        // 5. Collapse excessive inter-character spacing (char-by-char extraction artifact)
        // "P i z z a" → "Pizza" but preserve intentional word spacing
        result = result.replace(Regex("""(?<=[A-Za-zÀ-ÿ]) (?=[A-Za-zÀ-ÿ])(?= [A-Za-zÀ-ÿ] )"""), "")

        // 6. Normalize multiple spaces to single space
        result = result.replace(Regex(""" {2,}"""), " ")

        // 7. Normalize multiple newlines to max 2
        result = result.replace(Regex("""\n{3,}"""), "\n\n")

        // 8. Remove common page header/footer patterns
        result = removePageHeaders(result)

        return result.trim()
    }

    /**
     * Removes repetitive page header/footer patterns:
     * - Page numbers: "Page 1", "1 of 10", "- 1 -"
     * - Repeated document titles (first line that appears on multiple pages)
     */
    private fun removePageHeaders(text: String): String {
        var result = text

        // Remove standalone page number patterns
        val pageNumberPatterns = listOf(
            Regex("""^[\s]*[-–—]?\s*\d{1,3}\s*[-–—]?[\s]*$""", RegexOption.MULTILINE),
            Regex("""^[\s]*[Pp]age\s+\d+[\s]*$""", RegexOption.MULTILINE),
            Regex("""^[\s]*[Pp]ag\.?\s*\d+[\s]*$""", RegexOption.MULTILINE),
            Regex("""^[\s]*\d+\s*/\s*\d+[\s]*$""", RegexOption.MULTILINE),
            Regex("""^[\s]*\d+\s+of\s+\d+[\s]*$""", RegexOption.MULTILINE),
        )

        pageNumberPatterns.forEach { pattern ->
            result = pattern.replace(result, "")
        }

        return result
    }
}
```

Key points:
- Complete ligature map per D-05
- Soft hyphen handling with both Unicode soft hyphen and visible hyphen
- Inter-character spacing fix for char-by-char extraction artifacts
- Page header/footer removal (page numbers in various formats)
- Preserves page structure with `=== PAGINA N ===` markers (matching existing OCR format)
