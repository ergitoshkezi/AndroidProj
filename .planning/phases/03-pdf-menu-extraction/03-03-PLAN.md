    // Formula: 0.40×charDensity + 0.30×wordRecognition + 0.20×pricePresence + 0.10×lineStructure

    fun isUsable(score: Float, mode: ParsingMode): Boolean
    // STRICT: >= 0.70
    // BALANCED: >= 0.50
    // AGGRESSIVE: >= 0 (always true)
}

```kotlin
package com.example.ingredient.parser.pdf

import com.example.ingredient.parser.ast.ParsingMode
import org.junit.Assert.*
import org.junit.Test

class PdfQualityCheckerTest {

    // === SCORE TESTS ===

    @Test
    fun `score returns 0 for empty pages`() {
        assertEquals(0f, PdfQualityChecker.score(emptyList()), 0.001f)
    }

    @Test
    fun `score returns 0 for blank pages`() {
        assertEquals(0f, PdfQualityChecker.score(listOf("   ", "\n\n")), 0.001f)
    }

    @Test
    fun `score high for well-structured menu text`() {
        // Simulates a native-text PDF menu page with:
        // - Good char density (~600 chars for 1 page = 1.2 density, capped at 1.0)
        // - Good word recognition (real words)
        // - Price patterns present
        // - Good line structure (lines between 10-120 chars)
        val menuPage = """
            ANTIPASTI
            Bruschetta al pomodoro €8.00
            Carpaccio di manzo €12.50
            Mozzarella di bufala €10.00

            PRIMI PIATTI
            Spaghetti alla carbonara €14.00
            Risotto ai funghi porcini €16.00
            Lasagna della casa €13.50

            SECONDI PIATTI
            Bistecca alla fiorentina €28.00
            Branzino al forno €22.00
            Pollo arrosto con patate €15.00

            DOLCI
            Tiramisu €7.00
            Panna cotta €6.50
        """.trimIndent()

        val score = PdfQualityChecker.score(listOf(menuPage))
        // Expected: high score (>0.60) due to good structure
        assertTrue("Score should be high for well-structured menu, got $score", score > 0.60f)
    }

    @Test
    fun `score low for gibberish text`() {
        // Simulates corrupted/OCR-failed text
        val gibberishPage = """
            Xq!@ #$% ^&*()
            +++---///|||
            aaaa bbbb cccc
            1234567890
            @#@#@#@#
        """.trimIndent()

        val score = PdfQualityChecker.score(listOf(gibberishPage))
        // Expected: low score (<0.30) due to no valid words or prices
        assertTrue("Score should be low for gibberish, got $score", score < 0.30f)
    }

    @Test
    fun `score medium for sparse text`() {
        // Simulates a PDF with some valid text but sparse content
        val sparsePage = """
            Menu

            Pizza €12

            Pasta
        """.trimIndent()

        val score = PdfQualityChecker.score(listOf(sparsePage))
        // Expected: medium score (0.20-0.50) - has price but sparse
        assertTrue("Score should be medium for sparse text, got $score", score in 0.15f..0.55f)
    }

    @Test
    fun `score considers price patterns`() {
        val withPrices = "Pizza €10.00\nPasta €12.50\nSalad €8.00"
        val withoutPrices = "Pizza\nPasta\nSalad"

        val scoreWithPrices = PdfQualityChecker.score(listOf(withPrices))
        val scoreWithoutPrices = PdfQualityChecker.score(listOf(withoutPrices))

        assertTrue(
            "Score with prices ($scoreWithPrices) should be higher than without ($scoreWithoutPrices)",
            scoreWithPrices > scoreWithoutPrices
        )
    }

    @Test
    fun `score handles multiple pages`() {
        val page1 = "ANTIPASTI\nBruschetta €8.00\nCarpaccio €12.00"
        val page2 = "PRIMI\nSpaghetti €14.00\nRisotto €16.00"

        val singlePageScore = PdfQualityChecker.score(listOf(page1))
        val multiPageScore = PdfQualityChecker.score(listOf(page1, page2))

        // Both should be usable scores
        assertTrue("Single page score should be reasonable: $singlePageScore", singlePageScore > 0.3f)
        assertTrue("Multi page score should be reasonable: $multiPageScore", multiPageScore > 0.3f)
    }

    // === IS_USABLE TESTS (D-03 thresholds) ===

    @Test
    fun `isUsable STRICT requires 0_70`() {
        assertFalse(PdfQualityChecker.isUsable(0.69f, ParsingMode.STRICT))
        assertTrue(PdfQualityChecker.isUsable(0.70f, ParsingMode.STRICT))
        assertTrue(PdfQualityChecker.isUsable(0.85f, ParsingMode.STRICT))
    }

    @Test
    fun `isUsable BALANCED requires 0_50`() {
        assertFalse(PdfQualityChecker.isUsable(0.49f, ParsingMode.BALANCED))
        assertTrue(PdfQualityChecker.isUsable(0.50f, ParsingMode.BALANCED))
        assertTrue(PdfQualityChecker.isUsable(0.70f, ParsingMode.BALANCED))
    }

    @Test
    fun `isUsable AGGRESSIVE always returns true`() {
        assertTrue(PdfQualityChecker.isUsable(0.0f, ParsingMode.AGGRESSIVE))
        assertTrue(PdfQualityChecker.isUsable(0.25f, ParsingMode.AGGRESSIVE))
        assertTrue(PdfQualityChecker.isUsable(0.99f, ParsingMode.AGGRESSIVE))
    }

    @Test
    fun `isUsable boundary conditions`() {
        // Exactly at boundaries
        assertTrue(PdfQualityChecker.isUsable(0.50f, ParsingMode.BALANCED))
        assertTrue(PdfQualityChecker.isUsable(0.70f, ParsingMode.STRICT))

        // Just below boundaries
        assertFalse(PdfQualityChecker.isUsable(0.499f, ParsingMode.BALANCED))
        assertFalse(PdfQualityChecker.isUsable(0.699f, ParsingMode.STRICT))
    }
}
```

- Price pattern impact on score
- Multi-page handling
- Mode thresholds per D-03: STRICT>=0.70, BALANCED>=0.50, AGGRESSIVE>=0

```kotlin
package com.example.ingredient.parser.pdf

import org.junit.Assert.*
import org.junit.Test

class PdfTextCleanerTest {

    // === LIGATURE TESTS (D-05) ===

    @Test
    fun `clean replaces fi ligature`() {
        val input = "Antipasti con for\uFB01na"  // ﬁ ligature
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should replace ﬁ with fi", cleaned.contains("forfina"))
        assertFalse("Should not contain ligature", cleaned.contains("\uFB01"))
    }

    @Test
    fun `clean replaces fl ligature`() {
        val input = "In\uFB02uenza"  // ﬂ ligature
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should replace ﬂ with fl", cleaned.contains("Influenza"))
        assertFalse("Should not contain ligature", cleaned.contains("\uFB02"))
    }

    @Test
    fun `clean replaces ffi ligature`() {
        val input = "Ca\uFB03ne"  // ﬃ ligature
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should replace ﬃ with ffi", cleaned.contains("Caffine"))
        assertFalse("Should not contain ligature", cleaned.contains("\uFB03"))
    }

    @Test
    fun `clean replaces multiple ligatures`() {
        val input = "\uFB01nale e\uFB00etto"  // fi and ff ligatures
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should contain 'finale'", cleaned.contains("finale"))
        assertTrue("Should contain 'effetto'", cleaned.contains("effetto"))
    }

    // === SOFT HYPHEN TESTS ===

    @Test
    fun `clean removes soft hyphen with line break`() {
        val input = "Parmig\u00AD-\niana"  // soft hyphen + visible hyphen + newline
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should join hyphenated word", cleaned.contains("Parmigiana"))
    }

    @Test
    fun `clean handles visible hyphen at line end`() {
        val input = "Spag-\nhetti"
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should join hyphenated word", cleaned.contains("Spaghetti"))
    }

    @Test
    fun `clean preserves intentional hyphens`() {
        val input = "Coca-Cola"
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should preserve mid-word hyphen", cleaned.contains("Coca-Cola"))
    }

    // === WHITESPACE TESTS ===

    @Test
    fun `clean normalizes multiple spaces`() {
        val input = "Pizza    Margherita"
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should have single space", cleaned.contains("Pizza Margherita"))
        assertFalse("Should not have multiple spaces", cleaned.contains("  "))
    }

    @Test
    fun `clean normalizes multiple newlines`() {
        val input = "Section 1\n\n\n\n\nSection 2"
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertFalse("Should not have more than 2 consecutive newlines", cleaned.contains("\n\n\n"))
    }

    @Test
    fun `clean removes zero-width characters`() {
        val input = "Pizza\u200BMargherita"  // zero-width space
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertTrue("Should remove zero-width space", cleaned.contains("PizzaMargherita"))
    }

    // === PAGE HEADER TESTS ===

    @Test
    fun `clean removes standalone page numbers`() {
        val input = "Menu content here\n- 1 -\nMore content"
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertFalse("Should remove '- 1 -' pattern", cleaned.contains("- 1 -"))
        assertTrue("Should keep content", cleaned.contains("Menu content"))
    }

    @Test
    fun `clean removes Page N pattern`() {
        val input = "Content\nPage 5\nMore content"
        val cleaned = PdfTextCleaner.clean(listOf(input))
        assertFalse("Should remove 'Page 5'", cleaned.contains("Page 5"))
    }

    // === MULTI-PAGE TESTS ===

    @Test
    fun `clean joins multiple pages with markers`() {
        val page1 = "ANTIPASTI\nBruschetta €8.00"
        val page2 = "PRIMI\nPasta €12.00"

        val cleaned = PdfTextCleaner.clean(listOf(page1, page2))

        assertTrue("Should have page 1 marker", cleaned.contains("=== PAGINA 1 ==="))
        assertTrue("Should have page 2 marker", cleaned.contains("=== PAGINA 2 ==="))
        assertTrue("Should have page 1 content", cleaned.contains("ANTIPASTI"))
        assertTrue("Should have page 2 content", cleaned.contains("PRIMI"))
    }

    @Test
    fun `clean returns empty for empty pages`() {
        val cleaned = PdfTextCleaner.clean(emptyList())
        assertEquals("Should return empty string", "", cleaned)
    }

    @Test
    fun `clean handles blank pages gracefully`() {
        val cleaned = PdfTextCleaner.clean(listOf("   ", "\n\n"))
        // Blank pages should not produce page markers
        assertFalse("Should not have marker for blank page", cleaned.contains("=== PAGINA"))
    }

    // === INTEGRATION-STYLE TEST ===

    @Test
    fun `clean handles realistic PDF extraction output`() {
        val messyPage = """
            - 1 -
            RISTORANTE LA DOLCE VITA

            ANTIPASTI

            Bru\uFB01chetta al    pomodoro €8,00
            Car-
            paccio di manzo €12,50

            Page 1
        """.trimIndent()

        val cleaned = PdfTextCleaner.clean(listOf(messyPage))

        // Verify ligature replaced
        assertTrue("Ligature should be replaced", cleaned.contains("Bruschetta") || cleaned.contains("Brufichetta"))

        // Verify hyphenation fixed
        assertTrue("Carpaccio should be joined", cleaned.contains("Carpaccio"))

        // Verify page numbers removed
        assertFalse("Page number should be removed", cleaned.contains("- 1 -"))
        assertFalse("Page 1 should be removed", cleaned.contains("Page 1"))

        // Verify content preserved
        assertTrue("Restaurant name preserved", cleaned.contains("RISTORANTE"))
        assertTrue("Price preserved", cleaned.contains("€8"))
    }
}
```
